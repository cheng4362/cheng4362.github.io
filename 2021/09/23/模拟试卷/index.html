题目

1、查询课程平均分从高到低显示（现实任课老师）；（5 分）

SELECT t.Tname,AVG(sc.StuScore) FROM score sc INNER JOIN course c ON sc.Courseid = c.Courseid
INNER JOIN teacher t ON c.teacherid = t.Tid
GROUP BY sc.Courseid ORDER BY AVG(sc.StuScore) DESC

2、查询每门课程被选修的学生数；（5 分）

SELECT *,COUNT(Stuid) FROM score GROUP BY Courseid;

3、查询 “C04” 课程分数小于 60，按分数降序排列的同学学号；（10 分）

SELECT * FROM score WHERE Courseid = 'C04' AND StuScore < 60 ORDER BY StuScore DESC;

4、查询至少有一门课与学号为 “N00001” 的同学所学相同的同学的学号和姓名（10 分）

SELECT s.Sid, s.SName FROM student s INNER JOIN score sc ON s.Sid = sc.Stuid
WHERE sc.Courseid IN (SELECT Courseid FROM score WHERE Stuid = 'N00001')
GROUP BY S.Sid;

5、按各科平均成绩从低到高和及格率的百分数从高到低顺序；（10 分）

SELECT sc.Courseid,ROUND(AVG(StuScore),2),
(SUM (CASE WHEN StuScore BETWEEN 60 AND 100 THEN 1 ELSE 0 END)/COUNT (*))*100||'%' 及格率
FROM score sc
GROUP BY sc.Courseid;
